cmake_minimum_required(VERSION 3.14.2)
project(build-tools)

set(CMAKE_C_COMPILER_WORKS TRUE CACHE INTERNAL "")
set(CMAKE_CXX_COMPILER_WORKS TRUE CACHE INTERNAL "")
set(CMAKE_C_COMPILER_FORCED TRUE CACHE INTERNAL "")
set(CMAKE_CXX_COMPILER_FORCED TRUE CACHE INTERNAL "")

set(CMAKE_C_STANDARD 11)
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_EXE_LINKER_FLAGS "-static")

set(THREADS_PREFER_PTHREAD_FLAG ON CACHE BOOL "" FORCE)
set(CMAKE_USE_PTHREADS_INIT TRUE CACHE BOOL "" FORCE)
set(Threads_FOUND TRUE CACHE BOOL "" FORCE)
set(CMAKE_THREAD_LIBS_INIT "-lpthread" CACHE STRING "" FORCE)
set(CMAKE_HAVE_THREADS_LIBRARY TRUE CACHE BOOL "" FORCE)

set(TOOLS_VERSION 35.0.2)
set(SRC ${PROJECT_SOURCE_DIR}/submodules)

add_definitions(-D_FILE_OFFSET_BITS=64)

set(protobuf_BUILD_TESTS OFF CACHE BOOL "" FORCE)
set(protobuf_BUILD_SHARED_LIBS OFF CACHE BOOL "" FORCE)
set(protobuf_BUILD_PROTOC_BINARIES OFF CACHE BOOL "" FORCE)
set(protobuf_BUILD_LIBPROTOC ON CACHE BOOL "" FORCE)

if(DEFINED EXTRA_LIB_DIRS)
    link_directories(${EXTRA_LIB_DIRS})
endif()

add_subdirectory(submodules/boringssl EXCLUDE_FROM_ALL)
add_subdirectory(submodules/fmtlib EXCLUDE_FROM_ALL)
add_subdirectory(submodules/pcre EXCLUDE_FROM_ALL)
add_subdirectory(submodules/libpng EXCLUDE_FROM_ALL)
add_subdirectory(submodules/jsoncpp EXCLUDE_FROM_ALL)
add_subdirectory(submodules/expat/expat EXCLUDE_FROM_ALL)
add_subdirectory(submodules/protobuf EXCLUDE_FROM_ALL)
add_subdirectory(submodules/googletest EXCLUDE_FROM_ALL)

add_subdirectory(cmake)

if(TARGET libcutils)
    target_compile_definitions(libcutils PRIVATE PROP_NAME_MAX=32 _GNU_SOURCE)

    # 需要 NDK 头文件的源文件：单独添加包含路径和 Bionic 宏
    set(NDK_SOURCES
        ${PROJECT_SOURCE_DIR}/submodules/core/libcutils/ashmem-dev.cpp
        ${PROJECT_SOURCE_DIR}/submodules/core/libcutils/native_handle.cpp
        ${PROJECT_SOURCE_DIR}/submodules/core/libcutils/trace-dev.cpp
        ${PROJECT_SOURCE_DIR}/submodules/core/libcutils/trace-host.cpp
        ${PROJECT_SOURCE_DIR}/submodules/core/libcutils/strlcpy.c
    )

    foreach(src_file IN LISTS NDK_SOURCES)
        if(EXISTS "${src_file}")
            set_source_files_properties(${src_file}
                PROPERTIES
                COMPILE_FLAGS "-isystem ${NDK_SYSROOT}/usr/include"
                COMPILE_DEFINITIONS "__BIONIC__=1"
            )
        endif()
    endforeach()

    # native_handle.cpp 额外枚举定义（确保 __BIONIC__ 分支可用）
    set_source_files_properties(
        ${PROJECT_SOURCE_DIR}/submodules/core/libcutils/native_handle.cpp
        PROPERTIES COMPILE_DEFINITIONS "ANDROID_FDSAN_OWNER_TYPE_NATIVE_HANDLE=10;__BIONIC__=1"
    )
endif()
