set(CMAKE_CXX_COMPILER_FORCED TRUE CACHE INTERNAL "")

set(CMAKE_C_STANDARD 11)
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_EXE_LINKER_FLAGS "-static")

set(THREADS_PREFER_PTHREAD_FLAG ON CACHE BOOL "" FORCE)
set(CMAKE_USE_PTHREADS_INIT TRUE CACHE BOOL "" FORCE)
set(Threads_FOUND TRUE CACHE BOOL "" FORCE)
set(CMAKE_THREAD_LIBS_INIT "-lpthread" CACHE STRING "" FORCE)
set(CMAKE_HAVE_THREADS_LIBRARY TRUE CACHE BOOL "" FORCE)

set(TOOLS_VERSION 35.0.2)
set(SRC ${PROJECT_SOURCE_DIR}/submodules)

add_definitions(-D_FILE_OFFSET_BITS=64)

set(protobuf_BUILD_TESTS OFF CACHE BOOL "" FORCE)
set(protobuf_BUILD_SHARED_LIBS OFF CACHE BOOL "" FORCE)
set(protobuf_BUILD_PROTOC_BINARIES OFF CACHE BOOL "" FORCE)
set(protobuf_BUILD_LIBPROTOC ON CACHE BOOL "" FORCE)

if(DEFINED EXTRA_LIB_DIRS)
    link_directories(${EXTRA_LIB_DIRS})

# ========== 针对 libcutils 的精确修复 ==========
if(TARGET libcutils)
    # 1. 添加 NDK 头文件路径，并定义 Bionic 宏，激活 Android 相关代码
    if(DEFINED NDK_SYSROOT)
        target_include_directories(libcutils SYSTEM BEFORE PRIVATE ${NDK_SYSROOT}/usr/include)
    endif()
    target_compile_definitions(libcutils PRIVATE
        PROP_NAME_MAX=32
        __BIONIC__=1
        _GNU_SOURCE
    )

    # 2. 需要纯净 GNU 环境的源文件列表（移除 NDK 头文件路径）
    set(GNU_ONLY_SOURCES
        ${PROJECT_SOURCE_DIR}/submodules/core/libcutils/canned_fs_config.cpp
        ${PROJECT_SOURCE_DIR}/submodules/core/libcutils/fs_config.cpp
        ${PROJECT_SOURCE_DIR}/submodules/core/libcutils/properties.cpp
        ${PROJECT_SOURCE_DIR}/submodules/core/libcutils/android_get_control_file.cpp
        ${PROJECT_SOURCE_DIR}/submodules/core/libcutils/config_utils.cpp
        ${PROJECT_SOURCE_DIR}/submodules/core/libcutils/fs.cpp
        ${PROJECT_SOURCE_DIR}/submodules/core/libcutils/hashmap.cpp
        ${PROJECT_SOURCE_DIR}/submodules/core/libcutils/load_file.cpp
        ${PROJECT_SOURCE_DIR}/submodules/core/libcutils/multiuser.cpp
        ${PROJECT_SOURCE_DIR}/submodules/core/libcutils/partition_utils.cpp
        ${PROJECT_SOURCE_DIR}/submodules/core/libcutils/qtaguid.cpp
        ${PROJECT_SOURCE_DIR}/submodules/core/libcutils/record_stream.cpp
        ${PROJECT_SOURCE_DIR}/submodules/core/libcutils/socket_inaddr_any_server_unix.cpp
        ${PROJECT_SOURCE_DIR}/submodules/core/libcutils/threads.cpp
        ${PROJECT_SOURCE_DIR}/submodules/core/libcutils/uevent.cpp
    )

    foreach(src_file IN LISTS GNU_ONLY_SOURCES)
        if(EXISTS "${src_file}")
            set_source_files_properties(${src_file}
                PROPERTIES INCLUDE_DIRECTORIES ""
            )
        endif()
    endforeach()

    # 3. 为需要额外 Android 宏的文件补充定义（native_handle.cpp 已在 __BIONIC__ 下自动获得）
    set_source_files_properties(
        ${PROJECT_SOURCE_DIR}/submodules/core/libcutils/ashmem-dev.cpp
        ${PROJECT_SOURCE_DIR}/submodules/core/libcutils/trace-dev.cpp
        PROPERTIES COMPILE_DEFINITIONS "__ANDROID__"
    )
endif()
