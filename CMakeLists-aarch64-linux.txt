cmake_minimum_required(VERSION 3.14.2)

set(CMAKE_C_COMPILER_WORKS TRUE CACHE BOOL "Skip C compiler test" FORCE)
set(CMAKE_CXX_COMPILER_WORKS TRUE CACHE BOOL "Skip CXX compiler test" FORCE)
set(CMAKE_C_COMPILER_ABI_COMPILED TRUE CACHE BOOL "Skip C ABI detection" FORCE)
set(CMAKE_CXX_COMPILER_ABI_COMPILED TRUE CACHE BOOL "Skip CXX ABI detection" FORCE)

project(build-tools)

set(CMAKE_C_STANDARD 11)
set(CMAKE_CXX_STANDARD 20)

enable_language(ASM)

set(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} -fdiagnostics-color=always -fPIC -Wno-attributes -std=gnu11")
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -fdiagnostics-color=always -fPIC -Wno-attributes -std=gnu++2a")
set(CMAKE_EXE_LINKER_FLAGS "-static")

set(TOOLS_VERSION 35.0.2)
set(SRC ${PROJECT_SOURCE_DIR}/submodules)

add_definitions(-D_FILE_OFFSET_BITS=64 -D__ANDROID__=0)

set(THREADS_PREFER_PTHREAD_FLAG ON CACHE BOOL "Prefer -pthread flag" FORCE)
set(CMAKE_HAVE_THREADS_LIBRARY TRUE CACHE BOOL "Have threads library" FORCE)
set(CMAKE_USE_PTHREADS_INIT TRUE CACHE BOOL "Use pthreads" FORCE)
set(Threads_FOUND TRUE CACHE BOOL "Threads found" FORCE)
set(CMAKE_THREAD_LIBS_INIT "-lpthread" CACHE STRING "Thread library" FORCE)

set(ZLIB_FOUND TRUE CACHE BOOL "" FORCE)
set(ZLIB_LIBRARY /usr/aarch64-linux-gnu/lib/libz.a CACHE FILEPATH "" FORCE)
set(ZLIB_LIBRARIES ${ZLIB_LIBRARY} CACHE STRING "" FORCE)
set(ZLIB_INCLUDE_DIR /usr/aarch64-linux-gnu/include CACHE PATH "" FORCE)
mark_as_advanced(ZLIB_LIBRARY ZLIB_INCLUDE_DIR)

set(protobuf_BUILD_TESTS OFF CACHE BOOL "" FORCE)
set(protobuf_BUILD_SHARED_LIBS OFF CACHE BOOL "" FORCE)
set(protobuf_BUILD_PROTOC_BINARIES OFF CACHE BOOL "" FORCE)
set(protobuf_BUILD_LIBPROTOC ON CACHE BOOL "" FORCE)

add_subdirectory(submodules/boringssl EXCLUDE_FROM_ALL)
add_subdirectory(submodules/fmtlib EXCLUDE_FROM_ALL)
add_subdirectory(submodules/pcre EXCLUDE_FROM_ALL)
add_subdirectory(submodules/libpng EXCLUDE_FROM_ALL)
add_subdirectory(submodules/jsoncpp EXCLUDE_FROM_ALL)
add_subdirectory(submodules/expat/expat EXCLUDE_FROM_ALL)
add_subdirectory(submodules/protobuf EXCLUDE_FROM_ALL)
add_subdirectory(submodules/googletest EXCLUDE_FROM_ALL)

add_subdirectory(cmake)
