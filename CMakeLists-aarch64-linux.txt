# ============================================================================
# CMakeLists-aarch64-linux.txt
# 功能: 用于 aarch64 Linux 平台交叉编译 aapt2 的 CMake 配置文件
# 说明: 移除了所有 Android 特定逻辑，强制启用 ASM，静态链接
# ============================================================================

cmake_minimum_required(VERSION 3.14.2)
project(build-tools-linux-aarch64)

set(CMAKE_C_STANDARD 11)
set(CMAKE_CXX_STANDARD 20)

enable_language(ASM)

set(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} -fcolor-diagnostics -Wno-attributes")
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -fcolor-diagnostics -Wno-attributes")

# 强制线程支持，绕过 FindThreads 检测失败
set(THREADS_PTHREAD_ARG "0" CACHE STRING "Forced" FORCE)
set(CMAKE_HAVE_LIBC_PTHREAD ON CACHE BOOL "Forced" FORCE)
set(CMAKE_USE_PTHREADS_INIT ON CACHE BOOL "Forced" FORCE)
add_definitions(-DHAVE_PTHREAD=1)

# 直接提供 Threads::Threads 目标，避免 boringssl 的 find_package(Threads) 报错
if(NOT TARGET Threads::Threads)
    add_library(Threads::Threads INTERFACE IMPORTED)
    set_property(TARGET Threads::Threads PROPERTY INTERFACE_LINK_LIBRARIES "-pthread")
endif()

set(TOOLS_VERSION 35.0.2)
set(SRC ${PROJECT_SOURCE_DIR}/submodules)

add_definitions(-D_FILE_OFFSET_BITS=64)

set(protobuf_BUILD_TESTS OFF CACHE BOOL "" FORCE)
set(protobuf_BUILD_SHARED_LIBS OFF CACHE BOOL "" FORCE)
set(protobuf_BUILD_PROTOC_BINARIES OFF CACHE BOOL "" FORCE)
set(protobuf_BUILD_LIBPROTOC ON CACHE BOOL "" FORCE)

add_subdirectory(submodules/boringssl EXCLUDE_FROM_ALL)
add_subdirectory(submodules/fmtlib EXCLUDE_FROM_ALL)
add_subdirectory(submodules/pcre EXCLUDE_FROM_ALL)
add_subdirectory(submodules/libpng EXCLUDE_FROM_ALL)
add_subdirectory(submodules/jsoncpp EXCLUDE_FROM_ALL)
add_subdirectory(submodules/expat/expat EXCLUDE_FROM_ALL)
add_subdirectory(submodules/protobuf EXCLUDE_FROM_ALL)
add_subdirectory(submodules/googletest EXCLUDE_FROM_ALL)

add_subdirectory(cmake)
